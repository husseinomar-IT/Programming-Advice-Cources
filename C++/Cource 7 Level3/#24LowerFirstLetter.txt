string LowerFirstLetterOfEachWord(string txt)
{
	bool isFirstLetter = true;
	cout << "firstLetters of this string: \n";
	for (short i = 0; i < txt.length(); i++)
	{
		if (txt[i] != ' ' && isFirstLetter)
		{
			txt[i] = tolower(txt[i]);
		}
		isFirstLetter = (txt[i] == ' ' ? true : false);
		
	}
	return txt;

}
